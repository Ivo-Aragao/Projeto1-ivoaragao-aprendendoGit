aprendendo git.

git init  //cria um arquivo so é vizualizado quando vizualizar arquivos oculttos

crie um arquivo de texto

git status  //ver o status do arquivo

git add "meu codigo.txt" //adiciona o arquivo descrito

git add . //adiciona todos os arquivos de uma vez

git commit -m "commit incial" // cria commit: commit refere-se ao processo de tornar permanente um conjunto de alterações, ou seja, de efetivar as alterações

git config --global user.email "you@example.com" //configura o commit antes de usa-lo

git config --global user.name "your name" //configura o commit antes de usa-lo

//"git commit -m "commit incial"" so funciona se vc configurar primeiro

git push //envia seus arquivo para o repositorio que vc escolher, neste caso "github"



git remote add origin https://github.com/Ivo-Aragao/Projeto1-ivoaragao-aprendendoGit //defini o local do repositorio pra enviar os arquivos

git push --set-upstream origin master //para autorizar o acesso ao github
